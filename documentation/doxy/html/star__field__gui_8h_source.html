22  * no checking is done.

25  inline QQuaternion rotateToVector(const Vec& from, const Vec& to) {
26  // first, we need to get both the vector of rotation and then
27  // angle as a prelude to computing the rotation quaternion.
28  auto axis_of_rotation = from.cross(to);
29  auto angle = std::acos(from.dot(to));
30  auto cos2 = std::cos(angle/2.0);
31  auto sin2 = std::sin(angle/2.0);
32  auto w = cos2;
33  auto x = axis_of_rotation.vec[0] * sin2;
34  auto y = axis_of_rotation.vec[1] * sin2;
35  auto z = axis_of_rotation.vec[2] * sin2;
36  return QQuaternion {(float) w, (float) -x, (float) y, (float) z};
37  }
